program -> VD S EOF

VD -> VD
    | Vd
    | ;

Vd -> INT D SEMI

D -> D COMMA d
   | d

d -> ID ASSIGN INT_NUM
   | ID LSQUARE INT_NUM RSQUARE
   | ID

code_block -> s
            | LBRACE S RBRACE

S -> S s
   | s

s -> assign_s SEMI
  | control_s
  | read_write_s SEMI
  | SEMI

control_s -> if_s
                   | while_s
                   | do_while_s SEMI
                   | return_s SEMI

read_write_s -> read_s
                      | write_s

assign_s -> ID LSQUARE exp RSQUARE ASSIGN exp
                  | ID ASSIGN exp

if_s -> if_stmt
              | if_stmt ELSE code_block

if_stmt -> IF LPAR exp RPAR code_block

while_s -> WHILE LPAR exp RPAR code_block
do_while_s -> DO code_block WHILE LPAR exp RPAR

return_s -> RETURN

read_s -> READ LPAR ID RPAR
write_s -> WRITE LPAR exp RPAR

exp -> INT
     | ID
     | ID LSQUARE exp RSQUARE
     | NOT_OP exp
     | exp AND_OP exp
     | exp OR_OP exp
     | exp PLUS exp
     | exp MINUS exp
     | exp MUL_OP exp
     | exp DIV_OP exp
     | exp LT exp
     | exp GT exp
     | exp EQ exp
     | exp NOTEQ exp
     | exp LTEQ exp
     | exp GTEQ exp
     | exp SHL_OP exp
     | exp SHR_OP exp
     | exp ANDAND exp
     | exp OROR exp
     | LPAR exp RPAR
     | MINUS exp
